        // Liste over alle hovedsteder for å lage feilalternativer

        const allCapitals = capitalData.map(item => item.capital);

        // --- DOM Elementer ---

        const scoreP1Element = document.getElementById('score-p1');

        const scoreP2Element = document.getElementById('score-p2');

        const countryNameElement = document.getElementById('country-name');

        const answersAreaElement = document.getElementById('answers-area');

        const threatElement = document.getElementById('threat');

        const startButton = document.getElementById('start-button');

        const startButtonContainer = document.getElementById('start-button-container');

        const gameOverMessageElement = document.getElementById('game-over-message');

        const gameOverTitleElement = document.getElementById('game-over-title');

        const finalScoreElement = document.getElementById('final-score');

        const playAgainButton = document.getElementById('play-again-button');

        const gameContainer = document.querySelector('.game-container'); // For å beregne trusselposisjon

        // --- Spillstatus ---

        let scoreP1 = 0;

        let scoreP2 = 0;

        let currentQuestionData = null; // Holder { country, capital } for runden

        let currentAnswerOptions = []; // Holder de 4 alternativene som vises

        let remainingQuestions = []; // Kopi av data som kan endres

        let threatPosition = 0; // 0 = start, 5 = game over (juster etter behov)

        const maxThreat = 5;

        let gameOver = true; // Starter som "game over" før man trykker start

        let player1Answered = false;

        let player2Answered = false;

        let player1Choice = null; // Hvilket alternativ P1 valgte

        let player2Choice = null; // Hvilket alternativ P2 valgte

        let roundIncorrect = false; // Ble det svart feil i denne runden?

        let activeQuestion = false; // Er spørsmålet aktivt for svar?

        // --- Spillfunksjoner ---

        // Blander en array (Fisher-Yates shuffle)

        function shuffleArray(array) {

            for (let i = array.length - 1; i > 0; i--) {

                const j = Math.floor(Math.random() * (i + 1));

                [array[i], array[j]] = [array[j], array[i]];

            }

        }

        // Starter spillet

        function startGame() {

            scoreP1 = 0;

            scoreP2 = 0;

            threatPosition = 0;

            gameOver = false;

            activeQuestion = false; // Ikke aktiv før første spørsmål lastes

            updateScores();

            updateThreatVisual();

            gameOverMessageElement.classList.add('hidden'); // Skjul game over-melding

            startButtonContainer.classList.add('hidden'); // Skjul startknapp

            // Lag en kopi og bland spørsmålene for denne spillomgangen

            remainingQuestions = [...capitalData];

            shuffleArray(remainingQuestions);

            loadQuestion();

        }

        // Laster neste spørsmål

        function loadQuestion() {

            // Sjekk om det er flere spørsmål igjen

            if (remainingQuestions.length === 0) {

                endGame("Ingen flere spørsmål!");

                return;

            }

             // Tilbakestill svarstatus for runden

            player1Answered = false;

            player2Answered = false;

            player1Choice = null;

            player2Choice = null;

            roundIncorrect = false;

            // Hent neste spørsmål

            currentQuestionData = remainingQuestions.pop(); // Tar siste element fra blandet liste

            // Generer feilalternativer

            let distractors = [];

            while (distractors.length < 3) {

                const randomCapital = allCapitals[Math.floor(Math.random() * allCapitals.length)];

                // Sørg for at feilalternativet ikke er riktig svar og ikke allerede er valgt

                if (randomCapital !== currentQuestionData.capital && !distractors.includes(randomCapital)) {

                    distractors.push(randomCapital);

                }

            }

            // Kombiner riktig svar og feilalternativer, og bland dem

            currentAnswerOptions = [currentQuestionData.capital, ...distractors];

            shuffleArray(currentAnswerOptions);

            // Vis landnavnet

            countryNameElement.textContent = `Hva er hovedstaden i ${currentQuestionData.country}?`;

            // Vis svaralternativer

            answersAreaElement.innerHTML = ''; // Tøm forrige runde

            currentAnswerOptions.forEach((option, index) => {

                const button = document.createElement('button');

                button.classList.add('answer-option', 'p-4', 'bg-gray-600', 'hover:bg-gray-500', 'rounded-lg', 'text-lg', 'font-medium');

                button.textContent = option;

                button.dataset.capital = option; // Lagre hovedstaden i data-attributt

                button.dataset.index = index; // Lagre knappens indeks (0-3)

                answersAreaElement.appendChild(button);

            });

             activeQuestion = true; // Nå kan spillere svare

        }

        // Håndterer svar fra en spiller

        function handleAnswer(player, keyIndex) {

             if (gameOver || !activeQuestion) return; // Ikke gjør noe hvis spillet er over eller spørsmål inaktivt

            const targetIndex = keyIndex; // 0, 1, 2, 3

            const targetButton = answersAreaElement.querySelector(`[data-index="${targetIndex}"]`);

            if (!targetButton) return; // Fant ikke knappen (ugyldig tast?)

            const selectedCapital = targetButton.dataset.capital;

            if (player === 1 && !player1Answered) {

                player1Answered = true;

                player1Choice = targetButton;

                targetButton.classList.add('selected-p1'); // Vis visuell markering

                if (selectedCapital !== currentQuestionData.capital) {

                    roundIncorrect = true; // Marker at minst én svarte feil

                }

            } else if (player === 2 && !player2Answered) {

                player2Answered = true;

                player2Choice = targetButton;

                targetButton.classList.add('selected-p2'); // Vis visuell markering

                 if (selectedCapital !== currentQuestionData.capital) {

                    roundIncorrect = true; // Marker at minst én svarte feil

                }

            }

            // Sjekk om begge har svart

            if (player1Answered && player2Answered) {

                endRound();

            }

        }

        // Avslutter runden, evaluerer svar og går til neste, eller avslutter spillet

        function endRound() {

            activeQuestion = false; // Stopp flere svar for denne runden

            // Vis riktige/feile svar visuelt

             answersAreaElement.querySelectorAll('.answer-option').forEach(button => {

                const capital = button.dataset.capital;

                if (capital === currentQuestionData.capital) {

                    button.classList.add('correct'); // Marker riktig svar

                }

                // Fjern valgrammer og legg til feil-markering om nødvendig

                 button.classList.remove('selected-p1', 'selected-p2');

                 if (player1Choice === button && capital !== currentQuestionData.capital) button.classList.add('incorrect');

                 if (player2Choice === button && capital !== currentQuestionData.capital) button.classList.add('incorrect');

            });

            // Oppdater poeng basert på lagrede valg

            if (player1Choice && player1Choice.dataset.capital === currentQuestionData.capital) {

                scoreP1++;

            }

             if (player2Choice && player2Choice.dataset.capital === currentQuestionData.capital) {

                scoreP2++;

            }

            updateScores();

            // Flytt trusselen hvis noen svarte feil

            if (roundIncorrect) {

                threatPosition++;

                updateThreatVisual();

            }

            // Sjekk om spillet er over pga trusselen

            if (threatPosition >= maxThreat) {

                setTimeout(() => endGame("Trusselen nådde frem!"), 1500); // Litt forsinkelse for å vise resultat

            } else {

                // Gå til neste spørsmål etter en kort pause

                setTimeout(loadQuestion, 2000); // 2 sekunders pause

            }

        }

        // Oppdaterer poengsumvisningen

        function updateScores() {

            scoreP1Element.textContent = scoreP1;

            scoreP2Element.textContent = scoreP2;

        }

        // Oppdaterer trusselens posisjon visuelt

        function updateThreatVisual() {

            const containerHeight = gameContainer.clientHeight;

            // Beregn topp-posisjon basert på trusselstatus

            // Går fra 0% til f.eks. 80% av høyden for å unngå å dekke alt

            const topPercentage = (threatPosition / maxThreat) * 80;

            threatElement.style.top = `${topPercentage}%`;

             // Endre farge når den nærmer seg

            if (threatPosition >= maxThreat -1) {

                 threatElement.style.backgroundColor = '#c53030'; // Mørkere rød

            } else {

                 threatElement.style.backgroundColor = '#e53e3e'; // Standard rød

            }

        }

        // Avslutter spillet og viser resultat

        function endGame(reason) {

            gameOver = true;

            activeQuestion = false;

            let winnerMessage = "";

            if (scoreP1 > scoreP2) {

                winnerMessage = "Spiller 1 vinner!";

            } else if (scoreP2 > scoreP1) {

                winnerMessage = "Spiller 2 vinner!";

            } else {

                winnerMessage = "Det ble uavgjort!";

            }

            gameOverTitleElement.textContent = reason; // Viser hvorfor spillet sluttet

            finalScoreElement.textContent = `Sluttresultat: Spiller 1: ${scoreP1} - Spiller 2: ${scoreP2}. ${winnerMessage}`;

            gameOverMessageElement.classList.remove('hidden'); // Vis meldingen

            startButtonContainer.classList.remove('hidden'); // Vis startknapp-området igjen

            startButton.classList.add('hidden'); // Skjul selve startknappen

            playAgainButton.classList.remove('hidden'); // Vis spill igjen-knappen

        }

        // --- Event Listeners ---

        startButton.addEventListener('click', startGame);

        playAgainButton.addEventListener('click', startGame); // Samme funksjon for å starte på nytt

        // Lytt etter tastetrykk for svar

        document.addEventListener('keydown', (event) => {

            if (gameOver || !activeQuestion) return;

            const key = event.key;

            // Spiller 1 Taster: 1, 2, 3, 4 -> Index 0, 1, 2, 3

            if (key === '1') handleAnswer(1, 0);

            else if (key === '2') handleAnswer(1, 1);

            else if (key === '3') handleAnswer(1, 2);

            else if (key === '4') handleAnswer(1, 3);

            // Spiller 2 Taster: 7, 8, 9, 0 -> Index 0, 1, 2, 3

            else if (key === '7') handleAnswer(2, 0);

            else if (key === '8') handleAnswer(2, 1);

            else if (key === '9') handleAnswer(2, 2);

            else if (key === '0') handleAnswer(2, 3);

        });

        // Initialiser trusselposisjon ved last

         window.onload = () => {

            updateThreatVisual(); // Sett trusselen til startposisjon

            playAgainButton.classList.add('hidden'); // Skjul spill igjen-knappen initielt

         };

    </script>
